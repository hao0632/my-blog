import os

def find_identical_parts(file1, file2, min_chunk_size=64):
    identical_chunks = []
    current_chunk = None

    with open(file1, 'rb') as f1, open(file2, 'rb') as f2:
        pos = 0
        while True:
            chunk1 = f1.read(3145728)  
            chunk2 = f2.read(3145728)

            if not chunk1:
                if current_chunk:
                    identical_chunks.append(current_chunk)
                break

            chunk_len = min(len(chunk1), len(chunk2))
            i = 0

            while i < chunk_len:
                if chunk1[i] == chunk2[i]:
                    if current_chunk is None:
                        current_chunk = {'start': pos + i, 'length': 0}
                    current_chunk['length'] += 1
                else:
                    if current_chunk and current_chunk['length'] >= min_chunk_size:
                        identical_chunks.append(current_chunk)
                    current_chunk = None
                i += 1

            pos += chunk_len

    return identical_chunks

def compare_files(file1, file2):
    diff_count = 0

    with open(file1, 'rb') as f1, open(file2, 'rb') as f2:
        pos = 0
        while True:
            chunk1 = f1.read(4096)
            chunk2 = f2.read(4096)

            if not chunk1:
                break

            if chunk1 != chunk2:
                for i in range(len(chunk1)):
                    if chunk1[i] != chunk2[i]:
                        diff_count += 1

            pos += len(chunk1)

    return diff_count

import urllib.request
import sys

if sys.platform == 'win32':
    import io
    sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')

BASE58_ALPHABET = '123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz'

def base58_decode(cid):
    """Base58 解码"""
    result = 0
    for char in cid:
        result = result * 58 + BASE58_ALPHABET.index(char)
    
    bytes_length = (result.bit_length() + 7) // 8 or 1
    decoded = result.to_bytes(bytes_length, 'big')
    return decoded

def validate_cid_v0(cid):
    """验证是否为有效的 CIDv0"""
    if not cid.startswith('Qm'):
        return False
    if len(cid) != 46:
        return False
    try:
        decoded = base58_decode(cid)
        if len(decoded) != 34:
            return False
        if decoded[0] != 0x12 or decoded[1] != 0x20:
            return False
        return True
    except Exception:
        return False

def get_ipfs_gateway_url(cid, gateway="https://ipfs.io/ipfs/"):
    """生成 IPFS 网关访问链接"""
    return f"{gateway}{cid}"

def fetch_ipfs_content(cid, timeout=10):
    """尝试从 IPFS 网关获取内容"""
    url = get_ipfs_gateway_url(cid)
    try:
        response = urllib.request.urlopen(url, timeout=timeout)
        content = response.read(2000).decode('utf-8', errors='replace')
        return content
    except Exception as e:
        return f"Error: {str(e)[:50]}"

import hashlib
from base58 import b58decode

# ─── 1. CID → SHA-256 哈希值 ───────────────────────
cids = [
    "QmTXLuTnvmXF7hZzpLvpji74YB8pGzALmmbvYjZ133GsQT",
    "QmSzFpYD1ZW9ju9mKfNtc9fYWrkCUtdqCF5URGf3kEcZcm",

    "QmayG5Fa4UJEG7nyKVS4u4rvgs7YKJShgzbcsjRCx5VPm2",
]

target_hashes = []
for cid in cids:
    raw = b58decode(cid)              # 34 字节
    assert raw[:2] == b'\x12\x20'     # 校验 multihash 头
    target_hashes.append(raw[2:])     # 32 字节 SHA-256

# ─── 2. UnixFS Protobuf 封装常量 ──────────────────
# 4 字节文件的固定包裹: 0a 09 08 02 12 04 <4字节> 18 04
PREFIX = bytes.fromhex("0a0908021204")  # 6字节
SUFFIX = bytes.fromhex("1804")          # 2字节

import string
CHARSET = (string.ascii_letters + string.digits + "_{").encode()  # 按需调整字符集
assert len(CHARSET) == 64

found = {}
target_set = {h: i for i, h in enumerate(target_hashes)}

print(f"开始爆破，搜索空间 {len(CHARSET)**4:,} 次...")
for a in CHARSET:
    for b in CHARSET:
        for c in CHARSET:
            for d in CHARSET:
                content = bytes([a, b, c, d])                      # 4字节
                pb = PREFIX + content + SUFFIX                      # 12字节
                h = hashlib.sha256(pb).digest()
                if h in target_set:
                    idx = target_set[h]
                    found[idx] = content.decode()
                    print(f"  [{idx}] {cids[idx][:20]}… → {content!r}")
                    if len(found) == len(cids):
                        break
            if len(found) == len(cids):
                break
        if len(found) == len(cids):
            break
    if len(found) == len(cids):
        break

print("\n=== 结果 ===")
for i in range(len(cids)):
    print(f"{i+1}. {found.get(i, '<未找到>')}")

    assert raw[:2] == b'\x12\x20'     # 校验 multihash 头
    target_hashes.append(raw[2:])     # 32 字节 SHA-256

# ─── 2. UnixFS Protobuf 封装常量 ──────────────────
# 5 字节文件的固定包裹: 0a 0b 08 02 12 05 <5字节> 18 05
PREFIX = bytes.fromhex("0a0b08021205")  # 6字节
SUFFIX = bytes.fromhex("1805")          # 2字节
TAIL_BYTE = b"}"                        # 末位固定

import string
CHARSET = (string.ascii_letters + string.digits + "_{").encode()  # 按需调整字符集
assert len(CHARSET) == 64

found = {}
target_set = {h: i for i, h in enumerate(target_hashes)}

print(f"开始爆破，搜索空间 {len(CHARSET)**4:,} 次...")
for a in CHARSET:
    for b in CHARSET:
        for c in CHARSET:
            for d in CHARSET:
                content = bytes([a, b, c, d]) + TAIL_BYTE          # 5字节
                pb = PREFIX + content + SUFFIX                      # 13字节
                h = hashlib.sha256(pb).digest()
                if h in target_set:
                    idx = target_set[h]
                    found[idx] = content.decode()
                    print(f"  [{idx}] {cids[idx][:20]}… → {content!r}")
                    if len(found) == len(cids):
                        break
            if len(found) == len(cids):
                break
        if len(found) == len(cids):
            break
    if len(found) == len(cids):
        break

print("\n=== 结果 ===")
for i in range(len(cids)):
    print(f"{i+1}. {found.get(i, '<未找到>')}")
